class GeneratorAgent:
    def __init__(self, model_manager):
        # 3B fine-tuned model - optimized for 16GB VRAM constraint
        self.model = model_manager.get_model("generator")
        self.retriever = DocumentRetriever(hybrid=True)
        self.auditor = AuditorAgent()
    
    async def process(self, query: str) -> Response:
        # Auto-retrieve context from pgvector
        context = await self.retriever.search(query)
        
        # Route based on content type
        if self.is_table_query(context):
            sql_result = await self.sql_agent.generate(context)
            return self.format_response(sql_result)
        
        # Generate response with reflection loop
        response = await self.generate_with_reflection(
            query, context, max_retries=3
        )
        return response
class SQLCodeGenerator:
    def __init__(self):
        self.model = load_finetuned_model("sql-coder")
        self.schema_manager = SchemaManager()
    
    def generate_query(self, question: str, tables: List[str]):
        schema = self.schema_manager.get_schema(tables)
        prompt = self.build_prompt(question, schema)
        
        # Fine-tuned model generates SQL
        sql = self.model.generate(prompt, temperature=0.1)
        
        # Validate and execute
        if self.validate_sql(sql):
            results = self.execute_safe(sql)
            return self.format_results(results)
        
        return self.handle_error(sql)
class EvaluationFramework:
    def __init__(self):
        self.judge_model = ModelManager().get_model("judge")
        self.metrics = MetricsCollector()
    
    async def evaluate_response(self, response, ground_truth):
        # AI-as-judge evaluation
        rubric_score = await self.judge_model.score(
            response, ground_truth, rubric=self.rubric
        )
        
        # Calculate metrics
        metrics = {
            'accuracy': self.calculate_accuracy(response, ground_truth),
            'relevance': rubric_score['relevance'],
            'completeness': rubric_score['completeness'],
            'latency': response.processing_time
        }
        
        self.metrics.log(metrics)
        return metrics
